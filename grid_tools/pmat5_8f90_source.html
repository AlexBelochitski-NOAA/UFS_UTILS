653 subroutine paraframe(xp,yp,zp, xv,yv,zv)! [paraframe]
654 use pmat4, only: cross_product,normalized
655 implicit none
656 real(dp),dimension(3),intent(in ):: xp,yp,zp, zv
657 real(dp),dimension(3),intent(out):: xv,yv
658 real(dp) :: xpofv,ypofv,theta,ctheta,stheta
659 real(dp),dimension(3):: xq,yq
660 xpofv=dot_product(xp,zv)
661 ypofv=dot_product(yp,zv)
662 theta=atan2(ypofv,xpofv); ctheta=cos(theta); stheta=sin(theta)
663 xq=zv-zp; xq=xq-zv*dot_product(xq,zv); xq=normalized(xq)
664 yq=cross_product(zv,xq)
665 xv=xq*ctheta-yq*stheta
666 yv=xq*stheta+yq*ctheta
